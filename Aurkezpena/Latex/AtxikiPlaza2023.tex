function J2ODE!(du,u,p,t)
	r2 = x^2+y^2+z^2
	r = sqrt(r2)
	aux1 = -μ/(r*r2)
	F = 1.5 - 7.5*(z/r)^2
	G =  3 + F
	aux2 = ϵ*R^2/r2
	aux3 = aux1*(1+aux2*F)
	aux4 = aux1*(1+aux2*G)
	du[4]=aux3*x
	du[5]=aux3*y
	du[6]=aux4*z
	return nothing
end
